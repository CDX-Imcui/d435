apt安装报错，所以编译源码
git clone -b v2.25.0 https://github.com/IntelRealSense/librealsense.git
tar xvf librealsense-2.25.0.tar.gz
cd librealsense-2.25.0
# 安装依赖
sudo apt-get update && sudo apt-get upgrade && sudo apt-get dist-upgrade
sudo apt-get install libglfw3-dev libgl1-mesa-dev libglu1-mesa-dev

# 从libalsense根目录运行Intel Realsense权限脚本
./scripts/setup_udev_rules.sh
# 编译安装
mkdir build && cd build
如果机器有 NVIDIA GPU，还可以考虑加 -DBUILD_WITH_CUDA=true 来启用 CUDA 加速。
cmake ../ -DBUILD_EXAMPLES=true -DFORCE_RSUSB_BACKEND=true -DCMAKE_BUILD_TYPE=release -DBUILD_GRAPHICAL_EXAMPLES=FALSE -DBUILD_WITH_OPENMP=true
# 内存大运行这个
sudo make uninstall && make clean && make -j$(nproc) && sudo make install
# 内存小运行这个
sudo make uninstall && make clean && make && sudo make install
验证安装是否成功
cd librealsense/build/examples/C/depth/
./rs-depth
cd librealsense/build/examples/C/color/
